# 陪玩平台核心业务流程设计

> 基于 GoFrame v2 架构规范，业务逻辑位于 `app/play/internal/logic/`。
> 金额单位统一为"分"（BIGINT），比例为"百分比"（INT）。
> 核心写操作使用数据库事务确保一致性。

## 一、订单全生命周期流程

### 1.1 订单状态机

### 1.2 创建订单

**入口**：`POST /api/playapi/order/create`

**参数**：goods_id, quantity, coupon_member_id(可选), remark(可选)

**详细步骤**：

```
func CreateOrder(ctx, memberId, goodsId, quantity, couponMemberId, remark):
    // 1. 查商品信息
    goods = db.GetById(play_goods, goodsId)
    if goods == nil || goods.status != 1:
        return error("商品不存在或已下架")

    // 2. 查陪玩师信息
    coach = db.GetById(play_coach, goods.coach_id)
    if coach == nil || coach.status != 1:
        return error("陪玩师不可用")

    // 3. 查会员等级折扣
    member = db.GetById(play_member, memberId)
    memberLevel = db.GetById(play_member_level, member.member_level_id)
    discount = 100  // 默认无折扣
    if memberLevel != nil && memberLevel.status == 1:
        discount = memberLevel.discount  // 如 90 表示九折

    // 4. 计算订单金额
    totalAmount = goods.price * quantity
    discountAmount = totalAmount * (100 - discount) / 100  // 会员折扣金额

    // 5. 处理优惠券
    couponAmount = 0
    if couponMemberId > 0:
        couponMember = db.GetById(play_coupon_member, couponMemberId)
        if couponMember == nil || couponMember.member_id != memberId:
            return error("优惠券无效")
        if couponMember.use_status != 0:
            return error("优惠券已使用或已过期")
        if couponMember.expire_at < NOW():
            return error("优惠券已过期")

        coupon = db.GetById(play_coupon, couponMember.coupon_id)
        afterDiscount = totalAmount - discountAmount
        if afterDiscount < coupon.min_amount:
            return error("未达到优惠券使用门槛")

        // 计算优惠券抵扣金额
        if coupon.type == 1 || coupon.type == 3:  // 满减券 / 无门槛券
            couponAmount = min(coupon.face_value, afterDiscount)
        elif coupon.type == 2:  // 折扣券
            couponAmount = afterDiscount * (100 - coupon.face_value) / 100

    // 6. 计算实付金额
    payAmount = totalAmount - discountAmount - couponAmount
    if payAmount < 0:
        payAmount = 0

    // 7. 事务写入
    tx.Begin()
    order = {
        id:              snowflake.Generate(),
        order_no:        generateOrderNo(),  // 格式: PB + yyyyMMddHHmmss + 6位随机数
        member_id:       memberId,
        coach_id:        coach.id,
        shop_id:         coach.shop_id,
        goods_id:        goods.id,
        goods_title:     goods.title,       // 冗余快照
        goods_price:     goods.price,       // 冗余快照
        quantity:        quantity,
        total_amount:    totalAmount,
        discount_amount: discountAmount,
        coupon_amount:   couponAmount,
        pay_amount:      payAmount,
        coupon_member_id: couponMemberId,
        order_status:    0,  // 待支付
    }
    db.Insert(play_order, order)

    // 8. 锁定优惠券
    if couponMemberId > 0:
        db.Update(play_coupon_member, couponMemberId, {
            use_status: 1,
            use_at: NOW(),
            order_id: order.id,
        })
        db.Increment(play_coupon, coupon.id, "used_num", 1)

    tx.Commit()
    return order
```

### 1.3 订单支付

**三种支付方式统一入口**：`POST /api/playapi/payment/pay`

**参数**：order_id, pay_type(1=微信,2=支付宝,3=余额)

```
func PayOrder(ctx, memberId, orderId, payType):
    order = db.GetById(play_order, orderId)
    if order == nil || order.member_id != memberId:
        return error("订单不存在")
    if order.order_status != 0:
        return error("订单状态不允许支付")

    // 检查是否超时（30分钟）
    if NOW() - order.created_at > 30分钟:
        cancelOrder(order, "支付超时自动取消")
        return error("订单已超时取消")

    if payType == 3:  // 余额支付
        return payByBalance(ctx, order)
    else:  // 微信/支付宝
        return payByThirdParty(ctx, order, payType)
```

#### 1.3.1 余额支付

```
func payByBalance(ctx, order):
    tx.Begin()

    member = db.GetByIdForUpdate(play_member, order.member_id)  // 行锁
    if member.balance < order.pay_amount:
        tx.Rollback()
        return error("余额不足")

    // 1. 扣减余额
    newBalance = member.balance - order.pay_amount
    db.Update(play_member, member.id, {balance: newBalance})

    // 2. 写余额流水
    db.Insert(play_balance_log, {
        id:             snowflake.Generate(),
        member_id:      member.id,
        biz_type:       2,  // 消费
        change_amount:  -order.pay_amount,  // 负数
        before_balance: member.balance,
        after_balance:  newBalance,
        remark:         "订单消费: " + order.order_no,
    })

    // 3. 创建支付记录
    db.Insert(play_payment, {
        id:          snowflake.Generate(),
        order_id:    order.id,
        member_id:   member.id,
        payment_no:  generatePaymentNo(),
        pay_type:    3,
        pay_amount:  order.pay_amount,
        pay_status:  1,  // 成功
        pay_at:      NOW(),
    })

    // 4. 更新订单状态
    db.Update(play_order, order.id, {
        pay_type:     3,
        order_status: 1,  // 已支付
        pay_at:       NOW(),
    })

    tx.Commit()

    // 5. 异步触发：检查充值/下单活动
    go checkOrderActivity(member.id, order)

    return success
```

```
func payByThirdParty(ctx, order, payType):
    // 1. 创建支付记录（待支付状态）
    payment = {
        id:          snowflake.Generate(),
        order_id:    order.id,
        member_id:   order.member_id,
        payment_no:  generatePaymentNo(),
        pay_type:    payType,
        pay_amount:  order.pay_amount,
        pay_status:  0,  // 待支付
    }
    db.Insert(play_payment, payment)

    // 2. 调用第三方支付SDK
    if payType == 1:
        payParams = wechatPay.CreateOrder(payment.payment_no, order.pay_amount)
    elif payType == 2:
        payParams = alipay.CreateOrder(payment.payment_no, order.pay_amount)

    // 3. 更新订单支付方式
    db.Update(play_order, order.id, {pay_type: payType})

    return payParams  // 返回给前端唤起支付
```

#### 1.3.3 支付回调处理

```
func PayCallback(ctx, payType, callbackData):
    // 1. 验签
    if payType == 1:
        ok, data = wechatPay.VerifyCallback(callbackData)
    elif payType == 2:
        ok, data = alipay.VerifyCallback(callbackData)
    if !ok:
        return error("验签失败")

    // 2. 查支付记录
    payment = db.GetByPaymentNo(data.out_trade_no)
    if payment == nil:
        return error("支付记录不存在")
    if payment.pay_status == 1:
        return success  // 幂等处理，已处理过

    tx.Begin()

    // 3. 更新支付记录
    db.Update(play_payment, payment.id, {
        trade_no:         data.trade_no,  // 第三方交易号
        pay_status:       1,
        pay_at:           NOW(),
        callback_content: json(callbackData),
    })

    // 4. 更新订单状态
    order = db.GetById(play_order, payment.order_id)
    if order.order_status == 0:  // 仅待支付状态才更新
        db.Update(play_order, order.id, {
            order_status: 1,  // 已支付
            pay_at:       NOW(),
        })

    tx.Commit()

    // 5. 异步触发活动检测
    go checkOrderActivity(order.member_id, order)

    return success
```

```
func FinishOrder(ctx, coachId, orderId):

    if order == nil || order.coach_id != coachId:
        return error("订单不存在或不属于你")
    if order.order_status != 2:

    db.Update(play_order, order.id, {
        order_status: 3,  // 已完成
        finish_at:    NOW(),
    })

    // 2. 更新陪玩师接单数
    db.Increment(play_coach, coachId, "total_orders", 1)

    // 3. 更新商品销量
    db.Increment(play_goods, order.goods_id, "sales_num", order.quantity)

    // 4. 触发利润分成（见 2.1）
    calculateProfit(tx, order)

    // 5. 增加会员经验值
    addMemberExp(tx, order.member_id, order.pay_amount / 100)  // 每消费1元得1经验

    tx.Commit()

    // 6. 异步检查下单活动
    go checkOrderActivity(order.member_id, order)

    return success
```

### 1.6 用户取消订单

**入口**：`POST /api/playapi/order/cancel`

```
func CancelOrder(ctx, memberId, orderId, cancelReason):
    order = db.GetById(play_order, orderId)
    if order == nil || order.member_id != memberId:
        return error("订单不存在")

    // 仅待支付和已支付（未接单）可取消
    if order.order_status != 0 && order.order_status != 1:
        return error("当前状态不允许取消")

    tx.Begin()

    db.Update(play_order, order.id, {
        order_status:  4,  // 已取消
        cancel_at:     NOW(),
        cancel_reason: cancelReason,
    })

    // 已支付的需要退款
    if order.order_status == 1:
        processRefund(tx, order)

    // 释放优惠券
    if order.coupon_member_id > 0:
        db.Update(play_coupon_member, order.coupon_member_id, {
            use_status: 0,  // 恢复为未使用
            use_at:     nil,
            order_id:   0,
        })
        db.Decrement(play_coupon, couponId, "used_num", 1)

    tx.Commit()
    return success
```

```
func ProcessRefund(ctx, orderId, approve):
    order = db.GetById(play_order, orderId)
    if order.order_status != 5:
        return error("订单不在退款中状态")

    tx.Begin()

    if approve:
        db.Update(play_order, order.id, {
            order_status: 6,  // 已退款
            cancel_at:    NOW(),
        })
        processRefund(tx, order)
    else:
        // 拒绝退款，恢复进行中
        db.Update(play_order, order.id, {
            order_status: 2,  // 恢复进行中
        })

    tx.Commit()
    return success
```

### 1.9 退款执行（公共方法）

```
func processRefund(tx, order):
    payment = db.GetByOrderId(play_payment, order.id)

    if order.pay_type == 3:  // 余额支付 → 退回余额
        member = db.GetByIdForUpdate(play_member, order.member_id)
        newBalance = member.balance + order.pay_amount
        db.Update(play_member, member.id, {balance: newBalance})

        db.Insert(play_balance_log, {
            id:             snowflake.Generate(),
            member_id:      member.id,
            biz_type:       3,  // 退款
            change_amount:  order.pay_amount,  // 正数
            before_balance: member.balance,
            after_balance:  newBalance,
            remark:         "订单退款: " + order.order_no,
        })
    else:  // 微信/支付宝 → 原路退款
        if order.pay_type == 1:
            wechatPay.Refund(payment.trade_no, order.pay_amount)
        elif order.pay_type == 2:
            alipay.Refund(payment.trade_no, order.pay_amount)

    // 更新支付记录
    db.Update(play_payment, payment.id, {
        pay_status:    3,  // 已退款
        refund_at:     NOW(),
        refund_amount: order.pay_amount,
    })

    // 释放优惠券
    if order.coupon_member_id > 0:
        db.Update(play_coupon_member, order.coupon_member_id, {
            use_status: 0,
            use_at:     nil,
            order_id:   0,
        })
```

## 二、利润分成流程

### 2.1 分成计算（订单完成时触发）

**入口**：由 `FinishOrder` 内部调用（见 1.5 节第3步）

**分成规则**：
- 平台抽成 = pay_amount × platform_rate%（platform_rate 来自 `play_coach_level.commission_rate`）
- 店铺抽成 = (pay_amount - 平台抽成) × shop_rate%（shop_rate 来自 `play_shop.commission_rate`，无店铺则为 0）
- 陪玩师收入 = pay_amount - 平台抽成 - 店铺抽成

```
func CalcProfit(ctx, order):
    // 1. 查陪玩师及其等级
    coach = db.GetById(play_coach, order.coach_id)
    if coach == nil:
        return error("陪玩师不存在")

    // 2. 获取平台抽成比例
    platformRate = 20  // 默认 20%
    if coach.coach_level_id > 0:
        coachLevel = db.GetById(play_coach_level, coach.coach_level_id)
        if coachLevel != nil && coachLevel.status == 1:
            platformRate = coachLevel.commission_rate

    // 3. 获取店铺抽成比例
    shopRate = 0
    shopId = coach.shop_id
    if shopId > 0:
        shop = db.GetById(play_shop, shopId)
        if shop != nil && shop.status == 1:
            shopRate = shop.commission_rate

    // 4. 计算各方金额（单位：分）
    platformAmount = order.pay_amount * platformRate / 100
    afterPlatform  = order.pay_amount - platformAmount
    shopAmount     = afterPlatform * shopRate / 100
    coachAmount    = afterPlatform - shopAmount

    // 5. 事务写入
    tx.Begin()

    // 5a. 写入利润分成流水
    db.Insert(play_profit_log, {
        id:              snowflake.Generate(),
        order_id:        order.id,
        order_no:        order.order_no,
        pay_amount:      order.pay_amount,
        coach_id:        coach.id,
        shop_id:         shopId,
        platform_rate:   platformRate,
        platform_amount: platformAmount,
        shop_rate:       shopRate,
        shop_amount:     shopAmount,
        coach_amount:    coachAmount,
        settle_status:   0,  // 待结算
    })

    // 5b. 更新陪玩师累计收入和可提现余额
    db.Update(play_coach, coach.id, {
        income_total:   coach.income_total + coachAmount,
        income_balance: coach.income_balance + coachAmount,
    })

    tx.Commit()
    return success
```

### 2.2 结算流程（后台批量结算）

**入口**：`POST /api/play/profit/settle`（后台管理端）

**参数**：coach_id(可选，不传则全部结算), profit_log_ids(可选，指定结算)

```
func SettleProfit(ctx, coachId, profitLogIds):
    // 1. 查询待结算记录
    query = "settle_status = 0"
    if coachId > 0:
        query += " AND coach_id = " + coachId
    if len(profitLogIds) > 0:
        query += " AND id IN (" + join(profitLogIds, ",") + ")"

    logs = db.Select(play_profit_log, query)
    if len(logs) == 0:
        return error("无待结算记录")

    // 2. 按陪玩师分组汇总
    coachSettleMap = {}  // coach_id -> 总结算金额
    for log in logs:
        coachSettleMap[log.coach_id] += log.coach_amount

    tx.Begin()

    // 3. 批量更新结算状态
    for log in logs:
        db.Update(play_profit_log, log.id, {
            settle_status: 1,
            settle_at:     NOW(),
        })

    // 4. 扣减陪玩师可提现余额（结算 = 提现到银行卡/微信）
    for coachId, settleAmount in coachSettleMap:
        coach = db.GetByIdForUpdate(play_coach, coachId)  // 行锁
        if coach.income_balance < settleAmount:
            tx.Rollback()
            return error("陪玩师 " + coachId + " 可提现余额不足")
        db.Update(play_coach, coachId, {
            income_balance: coach.income_balance - settleAmount,
        })

    tx.Commit()
    return success
```

### 3.2 余额变动统一方法

```
func ChangeBalance(ctx, tx, memberId, bizType, bizId, changeAmount, remark):
    // 必须在事务内调用，tx 由调用方传入
    // changeAmount: 正数=增加，负数=减少

    member = db.GetByIdForUpdate(play_member, memberId)  // 行锁防并发

        member_id:      memberId,

        biz_id:         bizId,
        change_amount:  changeAmount,
        before_balance: member.balance,
        after_balance:  newBalance,
        remark:         remark,
    })

    return newBalance
```

### 3.3 充值支付回调处理

**入口**：`POST /api/playapi/recharge/callback/{payType}`

```
func RechargeCallback(ctx, payType, callbackData):
    // 1. 验签
    if payType == 1:
        ok, data = wechatPay.VerifyCallback(callbackData)
    elif payType == 2:
        ok, data = alipay.VerifyCallback(callbackData)
    if !ok:
        return error("验签失败")

    // 2. 查充值订单
    rechargeOrder = db.GetByOrderNo(play_recharge_order, data.out_trade_no)
    if rechargeOrder == nil:
        return error("充值订单不存在")
    if rechargeOrder.pay_status == 1:
        return success  // 幂等处理

    tx.Begin()

    // 3. 更新充值订单状态
    db.Update(play_recharge_order, rechargeOrder.id, {
        trade_no:   data.trade_no,
        pay_status: 1,  // 支付成功
        pay_at:     NOW(),
    })

    // 4. 增加余额（充值金额 + 赠送金额）
    totalAdd = rechargeOrder.amount + rechargeOrder.gift_amount
    ChangeBalance(ctx, tx, rechargeOrder.member_id,
        1,  // biz_type=充值
        rechargeOrder.id,
        totalAdd,
        "充值: " + rechargeOrder.order_no +
        " (充值" + rechargeOrder.amount + "+赠送" + rechargeOrder.gift_amount + ")")

    tx.Commit()

    // 5. 异步触发：检查充值活动
    go checkRechargeActivity(rechargeOrder.member_id, rechargeOrder.amount)

    return success
```

```
func ClaimCoupon(ctx, memberId, couponId):
    // 1. 查优惠券模板
    coupon = db.GetById(play_coupon, couponId)
    if coupon == nil || coupon.status != 1:
        return error("优惠券不存在或已关闭")

    // 2. 校验有效期
    now = NOW()
    if coupon.valid_type == 1:  // 固定时间段
        if now < coupon.valid_start_at || now > coupon.valid_end_at:
            return error("不在领取时间范围内")

    // 3. 校验总量
    if coupon.total_num > 0 && coupon.claim_num >= coupon.total_num:
        return error("优惠券已领完")

    // 4. 校验每人限领
    claimedCount = db.Count(play_coupon_member,
        "coupon_id = ? AND member_id = ?", couponId, memberId)
    if claimedCount >= coupon.per_limit:
        return error("已达领取上限")

    // 5. 校验新人券条件
    if coupon.is_new_member == 1:
        member = db.GetById(play_member, memberId)
        if coupon.new_member_days > 0:
            daysSinceRegister = (now - member.created_at).Days()
            if daysSinceRegister > coupon.new_member_days:
                return error("新人券仅注册" + coupon.new_member_days + "天内可领取")

    // 6. 计算过期时间
    if coupon.valid_type == 1:  // 固定时间段
        expireAt = coupon.valid_end_at
    elif coupon.valid_type == 2:  // 领取后N天
        expireAt = now + coupon.valid_days天

    // 7. 事务写入
    tx.Begin()

    db.Insert(play_coupon_member, {
        id:         snowflake.Generate(),
        coupon_id:  couponId,
        member_id:  memberId,
        use_status: 0,  // 未使用
        claim_at:   now,
        expire_at:  expireAt,
    })

    db.Increment(play_coupon, couponId, "claim_num", 1)

    tx.Commit()
    return success
```

```
func GetAvailableCoupons(ctx, memberId, orderAmount):
    // 查询该会员所有未使用且未过期的优惠券
    coupons = db.Select(play_coupon_member,
        "member_id = ? AND use_status = 0 AND expire_at > ?",
        memberId, NOW())

    result = []
    for cm in coupons:
        coupon = db.GetById(play_coupon, cm.coupon_id)
        if coupon == nil || coupon.status != 1:
            continue

        // 校验最低消费门槛
        if orderAmount < coupon.min_amount:
            continue

        // 计算优惠金额
        if coupon.type == 1 || coupon.type == 3:  // 满减券 / 无门槛券
            discountAmount = min(coupon.face_value, orderAmount)
        elif coupon.type == 2:  // 折扣券
            discountAmount = orderAmount * (100 - coupon.face_value) / 100

        result.append({
            coupon_member_id: cm.id,
            coupon:           coupon,
            discount_amount:  discountAmount,
            expire_at:        cm.expire_at,
        })

    // 按优惠金额降序排列，优先展示最优惠的
    sort(result, by: discount_amount DESC)
    return result
```

```
func JoinActivity(ctx, memberId, activityId):
    // 1. 查活动
    activity = db.GetById(play_activity, activityId)
    if activity == nil || activity.status != 1:
        return error("活动不存在或已关闭")
    if NOW() < activity.start_at || NOW() > activity.end_at:
        return error("活动不在有效期内")

    // 2. 检查参与人数上限
    if activity.max_num > 0 && activity.join_num >= activity.max_num:
        return error("活动参与人数已满")

    // 3. 检查是否已参与
    existing = db.GetOne(play_activity_join,
        "activity_id = ? AND member_id = ?", activityId, memberId)
    if existing != nil:
        return error("已参与该活动")

    // 4. 校验参与条件
    switch activity.condition_type:
        case 0:  // 无条件，直接参与
            break
        case 1:  // 需报名，直接写入（后续由后台审核）
            break
        case 2:  // 充值满额
            // 查询该会员累计充值金额
            totalRecharge = db.Sum(play_recharge_order,
                "member_id = ? AND pay_status = 1", memberId, "amount")
            if totalRecharge < activity.condition_value:
                return error("充值金额未达到活动要求")
        case 3:  // 下单满额
            totalOrders = db.Count(play_order,
                "member_id = ? AND order_status = 3", memberId)
            if totalOrders < activity.condition_value:
                return error("完成订单数未达到活动要求")
        case 4:  // 完成步骤（图文步骤活动）
            break  // 先报名，后续逐步完成

    // 5. 写入参与记录
    tx.Begin()

    joinStatus = 0  // 已报名
    if activity.condition_type == 0 || activity.condition_type == 2 || activity.condition_type == 3:
        joinStatus = 2  // 条件已满足，直接标记为已完成

    db.Insert(play_activity_join, {
        id:           snowflake.Generate(),
        activity_id:  activityId,
        member_id:    memberId,
        join_status:  joinStatus,
        current_step: 0,
        finish_at:    joinStatus == 2 ? NOW() : nil,
    })

    db.Increment(play_activity, activityId, "join_num", 1)

    tx.Commit()

    // 6. 自动发奖判断
    if joinStatus == 2 && activity.is_auto_reward == 1:
        go GrantActivityReward(ctx, activityId, memberId)

    return success
```

```
func GrantActivityReward(ctx, activityId, memberId):
    // 1. 查参与记录
    join = db.GetOne(play_activity_join,
        "activity_id = ? AND member_id = ?", activityId, memberId)
    if join == nil || join.join_status < 2:
        return error("活动未完成")
    if join.join_status == 3:
        return success  // 已领奖，幂等处理

    // 2. 查活动奖励配置
    rewards = db.Select(play_activity_reward,
        "activity_id = ?", activityId)
    if len(rewards) == 0:
        return error("活动无奖励配置")

    tx.Begin()

    // 3. 遍历奖励逐一发放
    for reward in rewards:
        switch reward.reward_type:
            case 1:  // 余额奖励
                ChangeBalance(ctx, tx, memberId,
                    4,  // biz_type=活动赠送
                    activityId,
                    reward.reward_value,  // 正数，单位分
                    "活动奖励: " + activityId)

            case 2:  // 优惠券奖励（reward_value = coupon_id）
                coupon = db.GetById(play_coupon, reward.reward_value)
                if coupon != nil && coupon.status == 1:
                    // 计算过期时间
                    if coupon.valid_type == 1:
                        expireAt = coupon.valid_end_at
                    elif coupon.valid_type == 2:
                        expireAt = NOW() + coupon.valid_days天

                    db.Insert(play_coupon_member, {
                        id:         snowflake.Generate(),
                        coupon_id:  coupon.id,
                        member_id:  memberId,
                        use_status: 0,
                        claim_at:   NOW(),
                        expire_at:  expireAt,
                    })
                    db.Increment(play_coupon, coupon.id, "claim_num", 1)

            case 3:  // 经验值奖励
                member = db.GetByIdForUpdate(play_member, memberId)
                newExp = member.exp + reward.reward_value
                db.Update(play_member, memberId, {exp: newExp})

                // 检查是否升级
                nextLevel = db.GetOne(play_member_level,
                    "min_exp <= ? AND status = 1 ORDER BY level DESC LIMIT 1", newExp)
                if nextLevel != nil && nextLevel.id != member.member_level_id:
                    db.Update(play_member, memberId, {
                        member_level_id: nextLevel.id,
                    })

            case 4:  // 会员等级天数奖励（延长VIP有效期）
                // reward_value = 天数
                // 业务含义：赠送N天高级会员体验
                // 具体实现取决于会员等级是否有有效期字段
                // 此处记录日志，由业务层扩展
                log.Info("赠送会员等级天数: memberId=%d, days=%d", memberId, reward.reward_value)

    // 4. 更新参与记录为已领奖
    db.Update(play_activity_join, join.id, {
        join_status: 3,  // 已领奖
        reward_at:   NOW(),
    })

    tx.Commit()
    return success
```

```
func CompleteActivityStep(ctx, memberId, activityId):
    // 1. 查活动（必须是图文步骤活动）
    activity = db.GetById(play_activity, activityId)
    if activity == nil || activity.type != 4:
        return error("活动不存在或非步骤活动")
    if activity.status != 1:
        return error("活动已关闭")
    if NOW() < activity.start_at || NOW() > activity.end_at:
        return error("活动不在有效期内")

    // 2. 查参与记录
    join = db.GetOne(play_activity_join,
        "activity_id = ? AND member_id = ?", activityId, memberId)
    if join == nil:
        return error("请先报名参与活动")
    if join.join_status >= 2:
        return error("活动已完成")

    // 3. 查总步骤数
    totalSteps = db.Count(play_activity_step,
        "activity_id = ?", activityId)

    // 4. 推进步骤
    newStep = join.current_step + 1

    tx.Begin()

    if newStep >= totalSteps:
        // 全部步骤完成
        db.Update(play_activity_join, join.id, {
            current_step: newStep,
            join_status:  2,  // 已完成
            finish_at:    NOW(),
        })
        tx.Commit()

        // 自动发奖
        if activity.is_auto_reward == 1:
            go GrantActivityReward(ctx, activityId, memberId)
    else:
        // 还有后续步骤
        db.Update(play_activity_join, join.id, {
            current_step: newStep,
            join_status:  1,  // 进行中
        })
        tx.Commit()

    return {current_step: newStep, total_steps: totalSteps, finished: newStep >= totalSteps}
```

```
        audit_status: auditStatus,
        audit_remark: auditRemark,
        audit_at:     NOW(),
    })

    // 2. 审核通过 → 创建陪玩师记录
    if auditStatus == 1:
        // 获取默认陪玩师等级（最低等级）
        defaultLevel = db.GetOne(play_coach_level,
            "status = 1 ORDER BY level ASC LIMIT 1")

        coach = {
            id:             snowflake.Generate(),
            member_id:      apply.member_id,
            coach_level_id: defaultLevel != nil ? defaultLevel.id : 0,
            shop_id:        0,  // 默认无店铺
            real_name:      apply.real_name,
            intro:          apply.skill_desc,
            total_orders:   0,
            total_score:    500,  // 默认5.00分
            score_num:      0,
            income_total:   0,
            income_balance: 0,
            is_online:      0,
            status:         1,
        }
        db.Insert(play_coach, coach)

        // 更新会员表
        db.Update(play_member, apply.member_id, {
            is_coach: 1,
        })

    tx.Commit()
    return success
```

## 七、评价系统

### 7.1 提交评价

**入口**：`POST /api/playapi/review/create`

**参数**：order_id, score(1-500，乘100存储), review_content, review_image(可选), is_anonymous(可选)

```
func CreateReview(ctx, memberId, orderId, score, reviewContent, reviewImage, isAnonymous):
    // 1. 校验订单
    order = db.GetById(play_order, orderId)
    if order == nil || order.member_id != memberId:
        return error("订单不存在")
    if order.order_status != 3:
        return error("订单未完成，无法评价")

    // 2. 检查是否已评价
    existingReview = db.GetOne(play_review, "order_id = ?", orderId)
    if existingReview != nil:
        return error("该订单已评价")

    // 3. 校验评分范围（100-500，即 1.00-5.00 分）
    if score < 100 || score > 500:
        return error("评分范围为 1-5 分")

    tx.Begin()

    // 4. 写入评价
    db.Insert(play_review, {
        id:             snowflake.Generate(),
        order_id:       orderId,
        member_id:      memberId,
        coach_id:       order.coach_id,
        score:          score,
        review_content: reviewContent,
        review_image:   reviewImage,
        is_anonymous:   isAnonymous,
        status:         1,  // 显示
    })

    // 5. 更新陪玩师评分
    coach = db.GetByIdForUpdate(play_coach, order.coach_id)
    newScoreNum = coach.score_num + 1
    // 重新计算平均分：(原总分 + 新评分) / 新评分人数
    // total_score 存储的是平均分（乘100），需要先还原总分
    oldTotalScore = coach.total_score * coach.score_num  // 还原历史总分
    newAvgScore = (oldTotalScore + score) / newScoreNum  // 新平均分

    db.Update(play_coach, order.coach_id, {
        total_score: newAvgScore,
        score_num:   newScoreNum,
    })

    tx.Commit()
    return success
```

## 八、定时任务汇总

| 任务名称 | 执行频率 | 方法 | 说明 |
|---------|---------|------|------|
| 订单超时取消 | 每分钟 | `CronCancelExpiredOrders()` | 扫描 order_status=0 且 created_at 超过30分钟的订单，自动取消并释放优惠券（见 1.10 节） |
| 优惠券过期处理 | 每天凌晨 00:05 | `CronExpireCoupons()` | 扫描 use_status=0 且 expire_at < NOW() 的会员优惠券，更新 use_status=2（见 4.3 节） |
| 活动过期处理 | 每天凌晨 00:10 | `CronExpireActivities()` | 扫描 end_at < NOW() 且 status=1 的活动，更新 status=0 |

```
func CronExpireActivities():
    // 每天凌晨 00:10 执行
    expiredActivities = db.Select(play_activity,
        "status = 1 AND end_at < ?", NOW())

    for activity in expiredActivities:
        db.Update(play_activity, activity.id, {
            status: 0,  // 关闭
        })

    log.Info("活动过期处理完成，共处理 %d 个", len(expiredActivities))
```
